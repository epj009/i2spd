(require 2htdp/image)

(define car-frame
(above (rectangle 30 10 "solid" "orange")
       (rectangle 50 15 "solid" "orange")))

(define wheels
  (beside (circle 8 "solid" "navy")
          (circle 8 "solid" "navy")))

(underlay/xy wheels
            -10 -20
            car-frame)
